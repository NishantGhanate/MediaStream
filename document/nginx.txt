uwsgi --http  :8000 --module cod.wsgi


sudo nano /etc/nginx/sites-available/cod.conf


# the upstream component nginx needs to connect to
upstream django {
    server unix:///home/nishant20/code/cod.sock;
}

# configuration of the server
server {
    listen      80;
    server_name 127.0.0.1  localhost;
    charset     utf-8;

    # max upload size
    client_max_body_size 5G;

    # Django media and static files
    location /media  {
        alias /home/nishant20/code/media;
    }
    location /static {
        alias /home/nishant20/code/static;
    }

    # Send all non-media requests to the Django server.
    location / {
        uwsgi_pass  django;
        include     /home/nishant20/code/uwsgi_params;
    }
}


sudo nano /home/nishant20/code/uwsgi_params

uwsgi_param  QUERY_STRING       $query_string;
uwsgi_param  REQUEST_METHOD     $request_method;
uwsgi_param  CONTENT_TYPE       $content_type;
uwsgi_param  CONTENT_LENGTH     $content_length;
uwsgi_param  REQUEST_URI        $request_uri;
uwsgi_param  PATH_INFO          $document_uri;
uwsgi_param  DOCUMENT_ROOT      $document_root;
uwsgi_param  SERVER_PROTOCOL    $server_protocol;
uwsgi_param  REQUEST_SCHEME     $scheme;
uwsgi_param  HTTPS              $https if_not_empty;
uwsgi_param  REMOTE_ADDR        $remote_addr;
uwsgi_param  REMOTE_PORT        $remote_port;
uwsgi_param  SERVER_PORT        $server_port;
uwsgi_param  SERVER_NAME        $server_name;


sudo ln -s /etc/nginx/sites-available/code.conf /etc/nginx/sites-enabled/


uwsgi --socket cod.sock --module cod.wsgi --chmod-socket=666
uwsgi --ini cod_uwsgi.ini
